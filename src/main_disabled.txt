#include <Arduino.h>
#include <WiFi.h>
#include <ESPAsyncWebServer.h>
#include <vector>
#include "config.h"

// ============ DATA STRUCTURES ============
struct Measurement {
  unsigned long timestamp;
  float pressure;
};

// ============ GLOBAL VARIABLES ============
AsyncWebServer server(80);
std::vector<Measurement> measurements;
SemaphoreHandle_t measurementMutex = NULL;

// ============ FORWARD DECLARATIONS ============
void sensorTask(void *parameter);
float readPressure();
float readRealPressure();
float simulatePressure();
void setupWiFi();
void setupWebServer();
String generateHTML();

// ============ SETUP ============
void setup() {
  Serial.begin(115200);
  delay(2000);  // Increased delay for serial initialization
  
  Serial.println("\n\n=== PneuNet Gripper Sensor System ===");
  Serial.flush();
  delay(100);
  
  // Create mutex for thread-safe data access
  Serial.println("Creating mutex...");
  Serial.flush();
  measurementMutex = xSemaphoreCreateMutex();
  delay(100);
  
  // Setup ADC
  Serial.println("Setting up ADC...");
  Serial.flush();
  analogReadResolution(12);  // 12-bit ADC (0-4095)
  Serial.println("Sensor initialized");
  if (SIMULATE_SENSOR) {
    Serial.println("SIMULATION MODE ENABLED");
  } else {
    Serial.println("Real sensor mode");
  }
  Serial.flush();
  delay(100);
  
  // Setup WiFi
  Serial.println("Setting up WiFi...");
  Serial.flush();
  setupWiFi();
  Serial.println("WiFi setup complete");
  Serial.flush();
  delay(500);
  
  // Setup Web Server
  Serial.println("Setting up Web Server...");
  Serial.flush();
  setupWebServer();
  Serial.println("Web Server setup complete");
  Serial.flush();
  delay(500);
  
  // Create FreeRTOS task for sensor reading
  Serial.println("Creating sensor task...");
  Serial.flush();
  xTaskCreatePinnedToCore(
    sensorTask,           // Function
    "SensorTask",         // Name
    8192,                 // Stack size
    NULL,                 // Parameter
    1,                    // Priority
    NULL,                 // Task handle
    0                     // Core 0
  );
  Serial.println("Sensor task created");
  Serial.flush();
  
  Serial.println("Setup complete!");
  Serial.printf("Connect to WiFi: %s\n", SSID);
  Serial.printf("Password: %s\n", PASSWORD);
  Serial.println("Open browser: http://192.168.4.1");
}

// ============ MAIN LOOP ============
void loop() {
  // Empty loop - server and sensor run on FreeRTOS tasks
  delay(1000);
}

// ============ SENSOR READING TASK ============
void sensorTask(void *parameter) {
  while (true) {
    float pressure = readPressure();
    unsigned long timestamp = millis();
    
    // Thread-safe data storage
    if (xSemaphoreTake(measurementMutex, portMAX_DELAY)) {
      if (measurements.size() < MAX_MEASUREMENTS) {
        measurements.push_back({timestamp, pressure});
      } else {
        // Remove oldest measurement when buffer full
        measurements.erase(measurements.begin());
        measurements.push_back({timestamp, pressure});
      }
      xSemaphoreGive(measurementMutex);
    }
    
    Serial.printf("Pressure: %.2f kPa, Time: %lu ms, Count: %zu\n", 
                  pressure, timestamp, measurements.size());
    
    delay(SAMPLING_DELAY);
  }
}

// ============ PRESSURE SENSOR READING ============
float readPressure() {
  if (SIMULATE_SENSOR) {
    return simulatePressure();
  } else {
    return readRealPressure();
  }
}

float readRealPressure() {
  // Read ADC (averaged over 10 samples for stability)
  int rawSum = 0;
  for (int i = 0; i < 10; i++) {
    rawSum += analogRead(ADC_PIN);
  }
  int adcValue = rawSum / 10;
  
  // Convert ADC value to voltage (12-bit ADC: 0-4095 maps to 0-3.3V)
  float Vout = adcValue * (3.3 / 4095.0);
  
  // MPX5010 datasheet formula: P(kPa) = (Vout/Vs - 0.04) / 0.09
  float pressure_kPa = (Vout / Vs - 0.04) / 0.09;
  
  // Clamp negative values (sensor noise at zero pressure)
  if (pressure_kPa < 0) pressure_kPa = 0;
  
  return pressure_kPa;
}

float simulatePressure() {
  static unsigned long startTime = millis();
  unsigned long elapsedTime = millis() - startTime;
  
  // Create oscillating pattern with noise
  float time_s = elapsedTime / 1000.0;
  float oscillation = SIM_AMPLITUDE * sin(2 * PI * SIM_FREQUENCY * time_s);
  float noise = (random(-100, 100) / 100.0) * SIM_NOISE;
  float pressure = SIM_BASE_PRESSURE + oscillation + noise;
  
  // Clamp to realistic values (0-10 kPa for demonstration)
  if (pressure < 0) pressure = 0;
  if (pressure > 10) pressure = 10;
  
  return pressure;
}

// ============ WIFI SETUP ============
void setupWiFi() {
  Serial.println("WiFi: Setting mode to AP...");
  Serial.flush();
  WiFi.mode(WIFI_AP);
  delay(100);
  
  Serial.println("WiFi: Starting softAP...");
  Serial.flush();
  bool result = WiFi.softAP(SSID, PASSWORD);
  Serial.printf("WiFi: softAP result = %d\n", result);
  Serial.flush();
  delay(100);
  
  IPAddress IP = WiFi.softAPIP();
  Serial.print("AP IP address: ");
  Serial.println(IP);
  Serial.flush();
}

// ============ WEB SERVER SETUP ============
void setupWebServer() {
  Serial.println("WebServer: Registering routes...");
  Serial.flush();
  
  // Serve main page
  server.on("/", HTTP_GET, [](AsyncWebServerRequest *request) {
    request->send(200, "text/html", generateHTML());
  });
  Serial.println("WebServer: Route / registered");
  Serial.flush();
  
  // API endpoint: get all measurements as JSON
  server.on("/api/data", HTTP_GET, [](AsyncWebServerRequest *request) {
    String json = "[";
    
    if (xSemaphoreTake(measurementMutex, portMAX_DELAY)) {
      for (size_t i = 0; i < measurements.size(); i++) {
        if (i > 0) json += ",";
        json += "{\"t\":" + String(measurements[i].timestamp) + 
                ",\"p\":" + String(measurements[i].pressure, 2) + "}";
      }
      xSemaphoreGive(measurementMutex);
    }
    
    json += "]";
    request->send(200, "application/json", json);
  });
  Serial.println("WebServer: Route /api/data registered");
  Serial.flush();
  
  // API endpoint: get latest measurement
  server.on("/api/latest", HTTP_GET, [](AsyncWebServerRequest *request) {
    String json = "{}";
    
    if (xSemaphoreTake(measurementMutex, portMAX_DELAY)) {
      if (!measurements.empty()) {
        Measurement latest = measurements.back();
        json = "{\"t\":" + String(latest.timestamp) + 
               ",\"p\":" + String(latest.pressure, 2) + "}";
      }
      xSemaphoreGive(measurementMutex);
    }
    
    request->send(200, "application/json", json);
  });
  Serial.println("WebServer: Route /api/latest registered");
  Serial.flush();
  
  // API endpoint: clear data
  server.on("/api/clear", HTTP_GET, [](AsyncWebServerRequest *request) {
    if (xSemaphoreTake(measurementMutex, portMAX_DELAY)) {
      measurements.clear();
      xSemaphoreGive(measurementMutex);
    }
    request->send(200, "application/json", "{\"status\":\"cleared\"}");
  });
  Serial.println("WebServer: Route /api/clear registered");
  Serial.flush();
  
  // API endpoint: export as CSV
  server.on("/api/export", HTTP_GET, [](AsyncWebServerRequest *request) {
    String csv = "Time_ms,Pressure_kPa\n";
    
    if (xSemaphoreTake(measurementMutex, portMAX_DELAY)) {
      for (const auto& m : measurements) {
        csv += String(m.timestamp) + "," + String(m.pressure, 4) + "\n";
      }
      xSemaphoreGive(measurementMutex);
    }
    
    request->send(200, "text/csv", csv);
  });
  Serial.println("WebServer: Route /api/export registered");
  Serial.flush();
  
  Serial.println("WebServer: Calling server.begin()...");
  Serial.flush();
  server.begin();
  Serial.println("WebServer: server.begin() completed");
  Serial.flush();
}
